country_event = {
	id = manchukuos.6
	title = manchukuos.6.t
	desc = manchukuos.6.d
	picture = GFX_handshake_russia2
	
	is_triggered_only = yes
	
	option = {
		name = manchukuos.6.a
		ai_chance = {
			factor = 100
		}
		if = {
			limit = {
				is_faction_leader = yes
			}
			SOV = {
				add_to_faction = MAN
			}
		}
		else = {
			SOV = {
				create_faction = russian_qing_alliance
				add_to_faction = MAN
			}
		}
		if = {
			limit = {
				MAN = {
					has_war_with = JAP
				}
			}
			SOV = {
				add_to_war = { 
					targeted_alliance = MAN 
					enemy = JAP
					hostility_reason = independence_war
				}
			}
		}
		if = {
			limit = {
				MAN = {
					has_war_with = event_target:WTT_current_china_leader
				}
			}
			custom_effect_tooltip = join_war_against_china_lord
			hidden_effect = {
				SOV = {
					add_to_war = { 
						targeted_alliance = MAN 
						enemy = event_target:WTT_current_china_leader
						hostility_reason = independence_war
					}
				}
			}
		}
		MAN = {
			add_ai_strategy = {
				type = alliance
				id = SOV
				value = 500
			}
		}
		SOV = {
			add_ai_strategy = {
				type = alliance
				id = MAN
				value = 200
			}
		}
		set_country_flag = no_alliance_with_china
	}
}
