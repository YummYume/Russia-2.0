		name = americas.5.a

			set_country_flag = fall_of_america
		}
		hidden_effect = {
			every_state = {
				limit = {
					is_core_of = USA
				}
				set_state_flag = USA_core_state
				SOV = {
					add_state_core = PREV
				}
			}
			every_state = {
				limit = {
					is_owned_by = USA
					NOT = {
						is_core_of = USA
					}
				}
				set_state_flag = USA_not_core_state
			}
		}
		USA = {
			every_unit_leader = {
				set_unit_leader_flag = flag_former_american_general
				set_nationality = SOV
			}
			transfer_navy = {
				target = SOV
			}
		}
		SOV = {
			annex_country = { target = USA transfer_troops = yes }
		}
		custom_effect_tooltip = SOV_soviet_american_cores
		hidden_effect = {
			news_event = { days = 1 id = newsrussia2.64 }
			SOV = {
				set_cosmetic_tag = SNAU
			}
		}
	}
}

#The War is Over - America Becomes Free
country_event = {
	id = americas.7
	title = americas.7.t
	desc = americas.7.d
	picture = GFX_report_event_generic_rally

	fire_only_once = yes

	trigger = {
		TAG = SOV
		SOV = {
			has_cosmetic_tag = SNAU
			has_country_flag = fall_of_america
			has_war = no
			is_subject = no
		}
		OR = {
			NOT = {
				country_exists = USA
			}
			USA = {
				is_subject_of = SOV
			}
		}
	}

	mean_time_to_happen = { days = 5 }

	immediate = {
		set_variable = {
			var = keep_union_stability_loss
			value = SOV.stability
		}
		divide_variable = {
			var = keep_union_stability_loss
			value = -2
		}
		set_variable = {
			var = keep_union_war_support_loss
			value = SOV.has_war_support
		}
		divide_variable = {
			var = keep_union_war_support_loss
			value = -2
		}
	}

	option = { #Release USA
		name = americas.7.a
		custom_effect_tooltip = SOV_america_release
		custom_effect_tooltip = SOV_america_cores
		hidden_effect = {
			if = {
				limit = {
					has_cosmetic_tag = SNAU
				}
				drop_cosmetic_tag = yes
			}

			if = {
				limit = {
					NOT = {
						country_exists = USA
					}
				}
				release = USA
			}

			else_if = {
				limit = {
					USA = {
						is_subject_of = SOV
					}
				}
				set_autonomy = {
					target = USA
					autonomy_state = autonomy_free
				}
			}

			if = {
				limit = {
					is_faction_leader = yes
					USA = {
						is_subject = no
						is_in_faction = no
					}
				}
				add_to_faction = USA
			}

			every_unit_leader = {
				limit = {
					has_unit_leader_flag = {
						flag = flag_former_american_general
					}
				}
				set_nationality = USA
			}		

			every_state = {
				limit = {
					has_state_flag = USA_core_state
				}
				remove_core_of = SOV
			}

			every_state = {
				limit = {
					has_state_flag = USA_not_core_state
				}
				USA = {
					transfer_state = PREV
				}
			}
		}
	}

	option = { #Keep the Union
		name = americas.7.b

		every_unit_leader = {
			limit = {
				has_unit_leader_flag = {
					flag = flag_former_american_general
				}
			}
			set_nationality = USA
		}
		add_stability = keep_union_stability_loss
		add_war_support = keep_union_war_support_loss
	}
}
