					}

						transfer_state = PREV
					}
				}

					country_event = { id = russia2_russia.412 hours = 6 }

	#Anarchist Civil War Caused by the Individualist, if the anarchist side win. Annex ROOT
	on_civil_war_end = {
		effect = {
			if = {
				limit = {
					ROOT = {
						has_government = anarchy
					}
					SOV = {
						has_government = anarchy
					}
				}
				SOV = {
					annex_country = {
						target = ROOT
						transfer_troops = yes
					}
				}
				clr_global_flag = an_anarchist_recolution_is_happening_indi
			}
			else_if = {
				limit = {
					ROOT = {
						NOT = {
							has_government = anarchy
						}
					}
					FROM = {
						has_government = anarchy
					}
					SOV = {
						has_government = anarchy
					}
				}
				clr_global_flag = an_anarchist_recolution_is_happening_indi
				ROOT = {
					SOV = {
						country_event = { id = russia2_russia.660 }
					}
				}
			}
		}
	}

	#Division Spawn in Targeted States after Declare War - Anarcho-Communism
	on_declare_war = {
		effect = {
			if = {
				limit ={
					ROOT = {
						tag = SOV
						has_government = anarchy
						has_country_flag = can_infiltrate_foreign_countries_wargoal
					}
					FROM = {
						any_owned_state = {
							has_state_flag = is_chosen_infiltrated_states_anarchist_communism
						}
					}
				}
				if = {
					limit = {
						ROOT = {
							NOT = {
								has_template = "Taynaya Militsiya Diviziya"
							}
						}
					}
					division_template = {
						name = "Taynaya Militsiya Diviziya"
						is_locked = yes
						regiments = {
							infantry = { x = 0 y = 0 }
							infantry = { x = 0 y = 1 }
							infantry = { x = 0 y = 2 }
							infantry = { x = 0 y = 3 }
							infantry = { x = 0 y = 4 }
							infantry = { x = 1 y = 0 }
						}
						support = {
							engineer = { x = 0 y = 0 }
						}
					}
				}
				FROM = {
					clear_variable = anarchist_state_infiltration
					hidden_effect = {
						country_event = { id = russia2_russia.657 hours = 8 }
					}
				}
			}
		}
	}

	#Remove Militia Division after A War
	on_peace = {
		effect = {
			if = {
				limit = {
					THIS = {
						tag = SOV
						has_government = anarchy
						has_country_flag = can_infiltrate_foreign_countries_wargoal
						has_template = "Taynaya Militsiya Diviziya"
					}
				}
				delete_unit_template_and_units = {
					division_template = "Taynaya Militsiya Diviziya"
					disband = no
				}
			}
		}
	}

	#Annex Puppet for Left Path Just in Case
	on_puppet = {
		effect = {
			if = {
				limit = {
					FROM = {
						tag = SOV
						has_government = anarchy
						NOT = {
							has_completed_focus = SOV_embrace_total_anarchy
						}
					}

						NOT = {
							tag = EAC
							tag = NAC
							tag = WAC
							tag = SWA
							tag = AEA
							tag = SAE
						}
					}
				}
				FROM = {
					annex_country = {
						target = ROOT
					}
				}
			}
		}
	}

	on_weekly_sov = {
		effect = {
			if = {
				limit = {
					has_government = anarchy
					NOT = {
						has_completed_focus = SOV_embrace_total_anarchy
					}
					any_country = {
						NOT = {
							tag = EAC
							tag = NAC
							tag = WAC
							tag = SWA
							tag = AEA
							tag = SAE
						}
						is_subject_of = SOV
					}
				}
				every_country = {
					limit = {
						is_subject_of = SOV
					}
					SOV = {
						annex_country = {
							target = PREV
						}
					}
				}
			}
		}
	}

	#Anarchist League Defensive
	on_war_relation_added = {
		effect = {
			if = {
				limit = {
					FROM = {
						OR = {
							tag = SOV
							is_in_faction_with = SOV
						}
					}
					ROOT = {
						NOT = {
							is_in_faction_with = SOV
							is_russian_state_russia2 = yes
						}
					}
					SOV = {
						has_completed_focus = SOV_we_are_stronger_together
						has_government = anarchy
						is_subject = no
						is_faction_leader = yes
					}
				}
				every_country = {
					limit = {
						OR = {
							tag = SOV
							is_in_faction_with = SOV
						}
						NOT = {
							has_war_with = ROOT
						}
					}
					ROOT = {
						declare_war_on = {
							target = PREV
							type = defense_anarchist_league
						}
					}
				}
			}
		}
	}
